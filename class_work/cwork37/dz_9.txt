4.	Вывести номера зачеток только студентов физико-технического факультета.
select n_z as "Номера зачеток Физико-технического факультета"
from zachetki
where id_studenta in
    (select id_st
    from exzamen
    where id_st in
        (select id_st
        from student
        where "group" in
            (select "group"
            from g
            where kafedra in
                (select kafedra
                from kafedra
                where decanat="Физико-технический"
                )
            )
        )
    )

5.	Выведите студентов физико-технического факультета сдавших иностранный язык на 5.
SELECT s.FIO, 
    (select e.ball
    from exzamen e
    where e.Id_St=s.ID_St and e.Predmet='Иностр. язык' and s."Group" in
            (select "group"
            from g
            where kafedra in
                (select kafedra 
                from kafedra
                where decanat='Физико-технический'
                )
            ) 
            and e.Ball=5
    ) as ball
from student s

7.	Напишите запрос, который выводит средний балл по экзаменам Васильевой.

12.	Выведите название кафедры студентов, не сдавших хотя бы один экзамен.
select kafedra
from kafedra k, g g, student s, exzamen e
where k.Kafedra=g.Kafedra and g."Group"=s."Group" and s.ID_St=e.Id_St and e.Ball is null

16.	Выведите перечень специальностей физико-математического факультета.
select s.special
from student s, g g, kafedra k
where s."Group"=g."Group" and g.Kafedra=k.Kafedra and k.Decanat="Физико-математический"
